交易API

交易API

在python代码中调用本地http api 实现实盘

import requests

## 在策略代码顶端添加本地server地址
REAL_URL = 'http://127.0.0.1:3001'

获取持仓信息

r = requests.get(REAL_URL + '/api/position')

## 持仓信息
print(r.json()['data'])

限价买单

r = requests.get(REAL_URL + '/api/buy', param = {
    stock: '002503',
    price: 2.62,
    amount: 100
})
请求返回
## 委托成功
{
    data: {
        order_id: '100'
    }
}

## 委托异常
{
    data: {
        message: '异常信息'
    }
}
